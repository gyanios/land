<p>앱이 열리지 않으면, Safari에서 열어주세요.</p>


<a id="safariBtn" class="btn hidden" href="#" target="_blank">Safari에서 열기</a>


<script>

  // 인스타그램 인앱브라우저 감지

  function isInstagramInApp() {

    return navigator.userAgent.includes("Instagram");

  }


  // 앱 열기 함수

  function openApp() {

    const isiOS = /iPhone|iPad|iPod/i.test(navigator.userAgent);

    const isAndroid = /Android/i.test(navigator.userAgent);


    if (isiOS) {

      window.location = "myapp://open"; // iOS 커스텀 스킴

    } else if (isAndroid) {

      window.location = "intent://open#Intent;scheme=myapp;package=com.example.myapp;end"; // Android intent

    }


    // 앱이 안 열리면 4초 후 스토어 이동

    setTimeout(() => {

      if (isiOS) {

        window.location = "https://apps.apple.com/app/idYOUR_APP_ID";

      } else if (isAndroid) {

        window.location = "https://play.google.com/store/apps/details?id=com.example.myapp";

      }

    }, 4000);

  }


  if (isInstagramInApp()) {

    // 인스타 브라우저면 Safari 안내 버튼 표시

    const btn = document.getElementById("safariBtn");

    btn.classList.remove("hidden");

    btn.href = window.location.href; // Safari에서 열기

  } else {

    // 일반 브라우저면 바로 앱 열기

    openApp();

  }

</script>
